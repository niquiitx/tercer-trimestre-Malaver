CREATE TABLE clientes (
id SERIAL PRIMARY KEY,
nombre VARCHAR(100) NOT NULL,
email VARCHAR(150) UNIQUE,
telefono VARCHAR(15),
fecha_registro DATE DEFAULT CURRENT_DATE,
activo BOOLEAN DEFAULT TRUE
);
